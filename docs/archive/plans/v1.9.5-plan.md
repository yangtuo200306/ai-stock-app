#### 5c. ask.py — 收集并保存 thinking steps

**文件：** `backend/app/api/ask.py` — `_stream_agent()` 函数

在 `_stream_agent()` 中收集 thinking 事件：

```python
def _stream_agent():
    full_answer = ""
    thinking_steps = []  # 收集 thinking 事件
    try:
        for sse_event in run_agent_loop(...):
            yield sse_event
            # 收集 thinking/tool_start/tool_done 事件
            try:
                data_str = sse_event
                if data_str.startswith("data: "):
                    data_str = data_str[6:]
                evt = json.loads(data_str.strip())
                if evt.get("type") in ("thinking", "tool_start", "tool_done"):
                    thinking_steps.append(evt)
                elif evt.get("type") == "text" and evt.get("content"):
                    full_answer += evt["content"]
            except Exception:
                pass
    finally:
        if full_answer:
            thinking_json = json.dumps(thinking_steps, ensure_ascii=False) if thinking_steps else None
            # 保存消息时传入 thinking_json
            _write_ask_message(..., thinking_json=thinking_json)
```
